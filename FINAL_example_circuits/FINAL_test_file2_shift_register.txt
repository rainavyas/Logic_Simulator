DEVICES{
switch1: SWITCH, initial 1;
switch2: SWITCH, initial 0;
switch3: SWITCH, initial 0;
D1: DTYPE;
D2: DTYPE;
D3: DTYPE;
D4: DTYPE;
clock: CLOCK, period 5;
}

CONNECT{
switch1 = D1.DATA;
D1.Q = D2.DATA;
D2.Q = D3.DATA;
D3.Q = D4.DATA;
clock = D1.CLK;
clock = D2.CLK;
clock = D3.CLK;
clock = D4.CLK;
switch2 = D1.SET;
switch2 = D2.SET;
switch2 = D3.SET;
switch2 = D4.SET;
switch3 = D1.CLEAR;
switch3 = D2.CLEAR;
switch3 = D3.CLEAR;
switch3 = D4.CLEAR;
}

MONITOR{
D1.Q;
D2.Q;
D3.Q;
D4.Q;
}

END
